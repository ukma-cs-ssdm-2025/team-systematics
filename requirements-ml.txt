# requirements-ml.txt
#
# ML (Machine Learning) бібліотеки для покращеної перевірки плагіату.
# Ці бібліотеки використовуються для семантичної аналізу текстів через
# нейронні мережі в модулі src/models/paraphrase.py.
#
# ⚠️ ВАЖЛИВО:
# - Ці бібліотеки ВЕЛИКІ (загалом ~2-3 GB) і встановлюються ДОВГО (~2+ хвилини)
# - Код має fallback механізм на TF-IDF (через scikit-learn), тому ML бібліотеки
#   НЕ ОБОВ'ЯЗКОВІ для роботи проєкту
# - Для швидких CI/CD тестів використовуйте тільки requirements-base.txt
# - ML бібліотеки потрібні тільки для покращеної (але не обов'язкової) перевірки плагіату
#
# Використання:
#   pip install -r requirements-base.txt -r requirements-ml.txt
#
# ============================================================================
# Deep Learning Framework
# ============================================================================
# PyTorch - фреймворк для глибокого навчання
# Розмір: ~2-3 GB, час встановлення: ~1-2 хвилини
torch==2.6.0

# ============================================================================
# Transformers & NLP Models
# ============================================================================
# Hugging Face Transformers - бібліотека для роботи з трансформерами

# Розмір: ~500 MB, час встановлення: ~30-60 секунд
transformers==4.41.2

# ============================================================================
# Sentence Embeddings
# ============================================================================

# Використовується в ParaphraseModel для перевірки плагіату
# Залежить від torch та transformers
# Розмір: ~100-200 MB, час встановлення: ~20-30 секунд
sentence-transformers==2.7.0
